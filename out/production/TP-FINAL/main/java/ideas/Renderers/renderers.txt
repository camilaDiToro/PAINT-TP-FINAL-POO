
void createButtons() {
    new MyButton((pt1, pt2, color, stroke, strokeWidth) -> new CircleRenderer(new Circle(pt1, pt2), color, stroke, strokeWidth));
    new MyButton((pt1, pt2, color, stroke, strokeWidth) -> new RectangleRenderer(new Rectangle(pt1, pt2), color, stroke, strokeWidth));
}


List<Renderer> renderers;

redraw() {
    // seleccionar
    for (renderer in renderers) {
        renderer.render(canvas)
    }
}

clickClick(Point pt1, Point pt2) {
    MyButton checked = getCheckedButton();
    Renderer renderer = checked.createRenderer(pt1, pt2)
    renderers.add(renderer);
    redraw()
}
